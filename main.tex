\begin{columns}[T]

  % --- Col 1: Introduction + Methods (stacked) ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Introduction}
      \begin{itemize}\setlength{\itemsep}{4pt}
        \item \textbf{What} --- A resolution gap separates ground-based
              rain radars (250~m, coastal only) from satellite products
              (IMERG~[2]: $\sim$8~km, global). No instrument covers the
              open ocean at sub-kilometer scale.

        \item \textbf{Why} --- The SWOT mission~[3] carries KaRIn, a
              Ka-band SAR at 35.75~GHz with \textbf{250~m/pixel}
              resolution. Ka-band backscatter is highly sensitive to
              atmospheric hydrometeors---an unexploited opportunity for
              precipitation estimation over the open ocean.

        \item \textbf{How} --- A machine learning framework trained on
              NEXRAD~[1] collocations extracts pixel-wise precipitation
              rates, validated with independent IMERG time series.
      \end{itemize}
    \end{clsblock}

    \vspace{-0.3cm}

    \begin{clsblock}{Methods}
      \textbf{Architecture.}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item U-Net encoder--decoder (depth=4, Swish activation,
              BatchNorm).
        \item Four input channels: incidence-corrected $\sigma_0$ (dB),
              total coherence, incidence angle, and model wind speed
              projected from the 2~km SWOT product.
        \item Output: pixel-wise precipitation rate (mm/h).
      \end{itemize}

      \vspace{0.1cm}
      \textbf{Training strategy.}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item Loss: regression MAE (99\%) + categorical MAE (1\%).
        \item Pixel weights decay linearly with distance to nearest
              NEXRAD station (weight 1 at 0~km, 0 at 225~km);
              null-DPR pixels down-weighted to 0.1\%.
        \item Quantile mapping aligns output distribution with NEXRAD.
      \end{itemize}

      \vspace{0.1cm}
      \textbf{Ensemble.}
      Five independently trained models; the consensus score
      (inter-model disagreement) provides a spatially-resolved
      proxy for prediction confidence.
    \end{clsblock}
  \end{column}

  % --- Col 2: Data ---
  \begin{column}{0.34\linewidth}
    \begin{clsblock}{Data}
      \begin{itemize}\setlength{\itemsep}{2pt}
        \item \textbf{7\,009} NEXRAD/SWOT collocated patches
              (512$\times$512~px at 250~m/pixel), Aug 2023--Feb 2025.
        \item \textbf{1\,090} patches contain precipitation
              ($>$1~mm/h on $>$1\% of pixels).
        \item Collocation constraints: $<$3~min temporal offset,
              $<$170~km range from a WSR-88D station.
        \item Ground truth: NEXRAD Digital Precipitation Rate (DPR),
              interpolated onto the SWOT grid.
      \end{itemize}

      \vspace{0.3cm}
      \noindent
      \begin{minipage}[t]{0.53\linewidth}
        \centering
        \includegraphics[width=\linewidth]{images/geo_distribution_us.png}
      \end{minipage}%
      \hfill
      \begin{minipage}[t]{0.43\linewidth}
        \centering
        \includegraphics[width=\linewidth]{images/geo_distribution_asia.png}
      \end{minipage}

      \vspace{0.2cm}
      \justifying
      Geographic distribution of collocations (pixel count per
      1$^\circ$ cell). \textbf{Left:} CONUS, Hawaii, Caribbean.
      \textbf{Right:} Korea, Japan.
    \end{clsblock}
  \end{column}

  % --- Col 3: Quantitative Evaluation ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Quantitative Evaluation}
      Categorical accuracy on three classes: [0--1), [1--10),
      and $\geq$10~mm/h. Spatially held-out test set (20\% of
      patches, never seen during training). Ensemble prediction
      is the mean of five independent models.

      \vspace{0.2cm}
      \centering
      \begin{tabular}{l@{\hspace{0.4cm}}ccc}
        \toprule
        & \multicolumn{3}{c}{\textbf{Predicted (mm/h)}} \\
        \cmidrule(lr){2-4}
        \textbf{NEXRAD} & {[0--1)} & {[1--10)} & {$\geq$10} \\
        \midrule
        {[0--1) mm/h}   & \textbf{94.1\%} & 20.9\% & 11.2\% \\
        {[1--10) mm/h}  & 5.3\% & \textbf{45.7\%} & 23.5\% \\
        {$\geq$10 mm/h} & 0.5\% & 33.4\% & \textbf{65.4\%} \\
        \bottomrule
      \end{tabular}

      \vspace{0.2cm}
      \justifying
      \begin{itemize}\setlength{\itemsep}{3pt}
        \item Rain-free areas: \textbf{94.1\%} accuracy. Heavy
              rainfall ($\geq$10~mm/h): \textbf{65.4\%}.
        \item Moderate rainfall ([1--10)~mm/h) is the hardest class,
              with 33.4\% misclassified as heavy rain---likely because
              moderate-to-heavy transitions are continuous.
        \item Ensemble reaches \textbf{68.4\%} accuracy, improving
              over individual models (63.6--68.0\%).
        \item Consensus score filters low-confidence predictions:
              retaining high-agreement pixels improves accuracy
              to $>$75\%.
        \item This 68\% accuracy is comparable to the consistency
              between two independent NEXRAD stations~[1], suggesting
              the model has reached the practical ceiling imposed by
              reference measurement uncertainty.
      \end{itemize}
    \end{clsblock}
  \end{column}

  % --- Col 3: Perspectives + References ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Perspectives}
      \begin{itemize}\setlength{\itemsep}{6pt}
        \item SWOT KaRIn observations at 250~m/pixel can be exploited
              to estimate precipitation rates over the open ocean,
              where no ground-based radar coverage exists.

        \item \textbf{Open question:} performance degrades at
              mid-latitudes (North Atlantic: r\,=\,$-$43\%),
              suggesting reduced sensitivity to stratiform
              precipitation. Further investigation is needed to
              characterize this limitation and identify the
              transition between convective and stratiform regimes.

        \item These results are consistent with C-band SAR rainfall
              retrievals from the Sentinel-1 constellation~[4].
              Integration into the standard SWOT unsmoothed product
              is under discussion, which would unlock precipitation
              data in data-sparse oceanic regions.
      \end{itemize}
    \end{clsblock}
