\begin{columns}[T]

  % --- Col 1: Introduction + Methods (stacked) ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Introduction}
      \textbf{What} --- A resolution gap separates ground-based rain
      radars (250~m, coastal only) from satellite products
      (IMERG~[2]: $\sim$8~km, global). No instrument covers the
      open ocean at sub-kilometer scale.

      \vspace{0.2cm}
      \textbf{Why} --- The SWOT mission~[3] carries KaRIn, a Ka-band
      SAR at 35.75~GHz with \textbf{250~m/pixel} resolution.
      Ka-band backscatter is highly sensitive to atmospheric
      hydrometeors---an unexploited opportunity for precipitation
      estimation over the open ocean.

      \vspace{0.2cm}
      \textbf{How} --- A machine learning framework trained on
      NEXRAD~[1] collocations extracts pixel-wise precipitation rates,
      validated with independent IMERG time series.
    \end{clsblock}

    \vspace{-0.3cm}

    \begin{clsblock}{Methods}
      \textbf{Architecture.}
      U-Net (depth=4, Swish, BatchNorm).
      Inputs: $\sigma_0$ (dB), coherence, incidence angle,
      model wind speed. Output: precipitation rate (mm/h).

      \vspace{0.2cm}
      \textbf{Training.}
      Loss: MAE (99\%) + categorical MAE (1\%).
      Pixel weights decay with distance to nearest NEXRAD station.
      Quantile mapping aligns output with NEXRAD.

      \vspace{0.2cm}
      \textbf{Ensemble.}
      Five independent models; consensus score provides prediction
      confidence.
    \end{clsblock}
  \end{column}

  % --- Col 2: Data ---
  \begin{column}{0.34\linewidth}
    \begin{clsblock}{Data}
      \textbf{7\,009} NEXRAD/SWOT collocated patches
      (512$\times$512~px at 250~m/pixel), Aug 2023--Feb 2025.
      \textbf{1\,090} contain precipitation
      ($>$1~mm/h on $>$1\% of pixels).
      Constraints: $<$3~min temporal offset,
      $<$170~km from a WSR-88D station.

      \vspace{0.3cm}

\begin{columns}[T]

  % --- Col 1: Quantitative Evaluation ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Quantitative Evaluation}
      Categorical accuracy on three classes: [0--1), [1--10),
      and $\geq$10~mm/h. Ensemble: \textbf{68.4\%} accuracy.

      \vspace{0.2cm}
      \centering
      \begin{tabular}{l@{\hspace{0.4cm}}ccc}
        \toprule
        & \multicolumn{3}{c}{\textbf{Predicted (mm/h)}} \\
        \cmidrule(lr){2-4}
        \textbf{NEXRAD} & {[0--1)} & {[1--10)} & {$\geq$10} \\
        \midrule
        {[0--1) mm/h}   & \textbf{94.1\%} & 20.9\% & 11.2\% \\
        {[1--10) mm/h}  & 5.3\% & \textbf{45.7\%} & 23.5\% \\
        {$\geq$10 mm/h} & 0.5\% & 33.4\% & \textbf{65.4\%} \\
        \bottomrule
      \end{tabular}

      \vspace{0.2cm}
      \justifying
      Rain-free: 94.1\%. Heavy rain ($\geq$10~mm/h): 65.4\%.
      Moderate rainfall is hardest (33.4\% confused with heavy).

      \vspace{0.2cm}
      Ensemble accuracy comparable to inter-station consistency
      of two independent NEXRAD radars~[1].
    \end{clsblock}
  \end{column}

  % --- Col 3: Perspectives ---
  \begin{column}{0.32\linewidth}
    \begin{clsblock}{Perspectives}
      \begin{itemize}\setlength{\itemsep}{6pt}
        \item SWOT KaRIn at 250~m/pixel can estimate precipitation
              over the open ocean, where no ground-based coverage exists.

        \item \textbf{Open question:} performance degrades at
              mid-latitudes (r\,=\,$-$43\%), suggesting reduced
              sensitivity to stratiform precipitation.

        \item Consistent with C-band SAR retrievals from
              Sentinel-1~[4]. Integration into the standard SWOT
              product would unlock precipitation data in data-sparse
              oceanic regions.
      \end{itemize}
    \end{clsblock}
