  % --- Col 1: Introduction & Methods ---
  \begin{column}{0.31\linewidth}
    \begin{clsblock}{Introduction \& Methods}
      \textbf{Gap:} Ground radars (NEXRAD) are coastal-only~[1].
      Satellite products (IMERG) are coarse ($\sim$8~km)~[2].
      Sentinel-1 SAR detects rain but has limited ocean coverage.

      \vspace{0.1cm}
      \textbf{SWOT KaRIn:} Ka-band interferometer (35.75~GHz),
      \textbf{250~m/pixel}, sensitive to hydrometeors,
      global ocean coverage~[3].

      \vspace{0.1cm}
      \textbf{Dataset:} 7\,009 NEXRAD/SWOT patches
      (512$\times$512~px, Aug 2023--Feb 2025),
      1\,090 with precipitation ($>$1~mm/h on $>$1\% of pixels).

      \vspace{0.1cm}
      \textbf{U-Net:} $\sigma_0$ + coherence + incidence angle
      + wind prior $\rightarrow$ precipitation rate (mm/h).
      Spatially weighted loss. Quantile mapping. Ensemble consensus.
    \end{clsblock}
  \end{column}

    \begin{clsblock}{Conclusions \& Perspectives}
      \begin{itemize}\setlength{\itemsep}{1pt}
        \item Ka-band precipitation at \textbf{250~m} demonstrated,
              effective in tropical regions
        \item Ensemble: \textbf{67\%} categorical accuracy;
              IMERG r\,=\,95\% (Pacific ITCZ)
        \item Open question: quality outside tropics ---
              convective vs.\ stratiform regime sensitivity
        \item Perspectives: Sentinel-1 C-band synergy~[4],
              standard SWOT product integration,
              data-sparse ocean coverage
      \end{itemize}
